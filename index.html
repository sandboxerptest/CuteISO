// ── Nick colour ──────────────────────────────────────────

function nickColor(nick) {

  let h = 0;

  for (let i = 0; i < nick.length; i++) h = (h * 31 + nick.charCodeAt(i)) & 0xffffffff;

  return `nc${Math.abs(h) % 8}`;

}


// ── Time ─────────────────────────────────────────────────

function ts() {

  const d = new Date();

  return `${String(d.getHours()).padStart(2,'0')}:${String(d.getMinutes()).padStart(2,'0')}`;

}


// ── Escape HTML ──────────────────────────────────────────

function esc(s) {

  return s.replace(/&/g,'&amp;').replace(/</g,'&lt;').replace(/>/g,'&gt;');

}


function stripCodes(text) {

  return text

    .replace(/\x03\d{1,2}(?:,\d{1,2})?/g, '')  // mIRC color codes \x03fg[,bg]

    .replace(/[\x02\x0f\x16\x1d\x1f]/g, '');    // bold, reset, reverse, italic, underline

}


function linkify(text) {

  return esc(stripCodes(text)).replace(/(https?:\/\/[^\s<>"]+)/g, '<a href="$1" target="_blank" rel="noopener">$1</a>');

}

}

// ── IRC message parser ───────────────────────────────────

// :prefix CMD [params] [:trailing]

function parseIRC(raw) {

  let prefix = '', cmd, params = [], trailing = '';

  let i = 0;

  if (raw[0] === ':') {

    const sp = raw.indexOf(' ');

    prefix = raw.slice(1, sp);

    raw = raw.slice(sp + 1);

  }

  if (raw.includes(' :')) {

    const idx = raw.indexOf(' :');

    trailing = raw.slice(idx + 2);

    raw = raw.slice(0, idx);

  }

  const parts = raw.trim().split(' ');

  cmd = parts[0].toUpperCase();

  params = parts.slice(1).filter(Boolean);

  if (trailing) params.push(trailing);

  const nick = prefix.split('!')[0];

  return { prefix, nick, cmd, params };

}

        }

        break;

      }


      const isSelf = nick === state.nick;

      addMessage(ch, {

        ts: ts(),

        nick,

        html: linkify(text),

        cls: isSelf ? 'self' : '',

        nc: isSelf ? '' : nickColor(nick),

      });

      break;

    }

      const text = params[1] || '';

      if (target.startsWith('#') || target.startsWith('&')) {

        ch = nick;   // private notice from a bot → open/use a query window

      } else {

        ch = null;   // server notice → server buffer

      }

      addMessage(ch, { ts: ts(), nick: nick || '●', html: linkify(text), cls: 'notice' });

      break;

    }


    // MODE

    case 'MODE': {

      const target = params[0];

      const modeStr = params.slice(1).join(' ');

      }

      break;

    }


    // LIST entry: 322 nick #channel usercount :topic

    case '322': {

      const name = params[1];

      const users = parseInt(params[2], 10) || 0;

      const topic = params[3] || '';

      break;

// ── WebSocket ────────────────────────────────────────────

function connectWS(opts) {

  // Use the specific bridge URL provided in the UI

  const ws = new WebSocket(opts.bridge);

  state.ws = ws;


  ws.onopen = () => {

    ws.send(JSON.stringify({

      type: 'connect',

      host: opts.host,

      port: opts.port,

      nick: opts.nick,

      username: opts.nick,

      realname: opts.realname || 'CuteISO',

      ssl: opts.ssl,

    }));

  };


  ws.onmessage = e => {

    const msg = JSON.parse(e.data);

    if (msg.type === 'connected') {

      state.connected = true;

      $statusDot.classList.add('online');

      $serverLabel.textContent = opts.host;

      $connectBtn.textContent = 'Disconnect';

      $connectBtn.classList.add('danger');

      $listBtn.disabled = false;

      addMessage(null, { ts: ts(), nick: '✓', html: `Piping to ${esc(opts.host)}`, cls: 'join' });

      if (opts.channel) state.autoJoin = opts.channel;

      showChat();

      switchChannel(null);

    } else if (msg.type === 'raw') {

      handleRaw(msg.data);

    } else if (msg.type === 'error') {

      addMessage(null, { ts: ts(), nick: '✖', html: esc(msg.data), cls: 'notice' });

    } else if (msg.type === 'disconnected') {

      onDisconnect(msg.data);


    // ── DCC ────────────────────────────────────────────

    } else if (msg.type === 'dcc_offer') {

      handleDccOffer(msg);

    } else if (msg.type === 'dcc_progress') {

      handleDccProgress(msg);

    } else if (msg.type === 'dcc_data') {

      // Collect the file chunks in memory

      if (!state.dccBuffers[msg.session_id]) {

        state.dccBuffers[msg.session_id] = [];

      }

      // Decode base64 to binary string, then save as Uint8Array

      const binaryString = atob(msg.data);

      const len = binaryString.length;

      const bytes = new Uint8Array(len);

      for (let i = 0; i < len; i++) {

        bytes[i] = binaryString.charCodeAt(i);

      }

      state.dccBuffers[msg.session_id].push(bytes);

    } else if (msg.type === 'dcc_complete') {

      handleDccComplete(msg);

      // Compile the chunks into a Blob and trigger local download

      if (state.dccBuffers[msg.session_id]) {

        const blob = new Blob(state.dccBuffers[msg.session_id]);

        const url = URL.createObjectURL(blob);

        const a = document.createElement('a');

        a.href = url;

        a.download = msg.filename;

        document.body.appendChild(a);

        a.click();

        document.body.removeChild(a);

        URL.revokeObjectURL(url);

        delete state.dccBuffers[msg.session_id]; // Free memory

      }

    } else if (msg.type === 'dcc_error') {

      handleDccError(msg);

    } else if (msg.type === 'dcc_chat_open') {

      handleDccChatOpen(msg);

    } else if (msg.type === 'dcc_chat_msg') {

      handleDccChatMsg(msg);

    } else if (msg.type === 'dcc_chat_closed') {

      handleDccChatClosed(msg);

    }

  };


  ws.onclose = () => onDisconnect('WebSocket closed');

  ws.onerror = () => onDisconnect('WebSocket error');

}


function onDisconnect(reason) {

  state.connected = false;

  state.ws = null;

  state.dccChats = {};

  state.activeDccSession = null;

  $statusDot.classList.remove('online');

  $serverLabel.textContent = 'Not connected';

  $connectBtn.textContent = 'Connect';

  $connectBtn.classList.remove('danger');

  $listBtn.disabled = true;

  closeBrowser();

  addMessage(null, { ts: ts(), nick: '✖', html: esc(reason || 'Disconnected'), cls: 'notice' });

}


function sendRaw(line) {

  sendWs({ type: 'send', data: line });

}


// ── Input handling ───────────────────────────────────────

function handleInput() {

  const val = $msgInput.value.trim();

  if (!val) return;

  $msgInput.value = '';


  // If we're in a DCC CHAT session, relay text directly to it

  if (state.activeDccSession) {

    const s = state.dccChats[state.activeDccSession];

    if (s && !s.closed) {

      const m = { ts: ts(), nick: state.nick, html: esc(val), cls: 'self' };

      s.messages.push(m);

      renderMessage(m);

      $messages.scrollTop = $messages.scrollHeight;

      sendWs({ type: 'dcc_chat_send', session_id: state.activeDccSession, text: val });

    }

    return;

  }


  if (val.startsWith('/')) {

    const parts = val.slice(1).split(' ');

    const cmd = parts[0].toUpperCase();

    const rest = parts.slice(1).join(' ');


    switch (cmd) {

      case 'JOIN':

        sendRaw(`JOIN ${rest}`);

        break;

      case 'PART':

        sendRaw(`PART ${state.activeChannel || rest}`);

        break;

      case 'NICK':

        sendRaw(`NICK ${rest}`);

        state.nick = rest;

        break;

      case 'ME':

        if (state.activeChannel) {

          sendRaw(`PRIVMSG ${state.activeChannel} :\x01ACTION ${rest}\x01`);

          addMessage(state.activeChannel, { ts: ts(), nick: `* ${state.nick}`, html: linkify(rest), cls: 'action' });

        }

        break;

      case 'MSG':

      case 'QUERY': {

        const [target, ...msgParts] = rest.split(' ');

        const text = msgParts.join(' ');

        sendRaw(`PRIVMSG ${target} :${text}`);

        addMessage(target, { ts: ts(), nick: state.nick, html: linkify(text), cls: 'self' });

        switchChannel(target);

        break;

      }

      case 'TOPIC':

        sendRaw(`TOPIC ${state.activeChannel} :${rest}`);

        break;

      case 'KICK': {

        const [kickNick, ...reason] = rest.split(' ');

        sendRaw(`KICK ${state.activeChannel} ${kickNick} :${reason.join(' ')}`);

        break;

      }

      case 'LIST':

        openBrowser();

        break;

      case 'QUOTE':

      case 'RAW':

        sendRaw(rest);

        break;

      default:

        sendRaw(val.slice(1)); // pass through unknown commands

    }

  } else {

    // Normal message to active channel

    const target = state.activeChannel;

    if (!target) {

      addMessage(null, { ts: ts(), nick: '●', html: 'Pipe into a channel first.', cls: 'system' });

      return;

    }

    sendRaw(`PRIVMSG ${target} :${val}`);

    addMessage(target, { ts: ts(), nick: state.nick, html: linkify(val), cls: 'self' });

  }

}

function fmtSize(n) {

  if (!n) return '';

  if (n < 1024) return `${n} B`;

  if (n < 1024 * 1024) return `${(n/1024).toFixed(1)} KB`;

  return `${(n/1024/1024).toFixed(2)} MB`;

}


// Pending banner offer (one at a time shown; queue the rest)

const dccQueue = [];

let dccBannerActive = null;


function showDccBanner(msg) {

  dccQueue.push(msg);

  if (!dccBannerActive) pumpDccBanner();

}


function pumpDccBanner() {

  if (!dccQueue.length) { dccBannerActive = null; return; }

  dccBannerActive = dccQueue.shift();

  const m = dccBannerActive;

  const isSend = m.kind === 'SEND';

  document.getElementById('dcc-b-title').textContent =

    isSend ? `📥 Piping file from ${m.nick}` : `💬 Piping chat from ${m.nick}`;

  document.getElementById('dcc-b-meta').textContent =

    isSend ? `${m.filename}${m.size ? '  ·  ' + fmtSize(m.size) : ''}` : 'FServ / direct chat session';

  document.getElementById('dcc-banner').classList.add('show');

  showChat();

}


function dismissBanner() {

  document.getElementById('dcc-banner').classList.remove('show');

  dccBannerActive = null;

  setTimeout(pumpDccBanner, 150);

}


document.getElementById('dcc-b-accept').addEventListener('click', () => {

  if (!dccBannerActive) return;

  sendWs({ type: 'dcc_accept', session_id: dccBannerActive.session_id });

  dismissBanner();

});


document.getElementById('dcc-b-decline').addEventListener('click', () => {

  if (!dccBannerActive) return;

  sendWs({ type: 'dcc_decline', session_id: dccBannerActive.session_id });

  dismissBanner();

});

    el.querySelector('.dcc-status').textContent =

      `${fmtSize(msg.received)} / ${fmtSize(msg.total)} (${pct}%)`;

  }

}

function openBrowser() {

  $browserOverlay.classList.add('open');

  if (state.browser.entries.length === 0 && !state.browser.loading) {

    requestList();

  }

}


function closeBrowser() {

  $browserOverlay.classList.remove('open');

}


function requestList() {

  state.browser.entries = [];

  state.browser.loading = true;

  $browserStatus.textContent = 'Piping channels…';

  $browserTbody.innerHTML = '';

  $browserEmpty.style.display = 'none';

  sendRaw('LIST');

}


function renderBrowser() {

  const { entries, sortCol, sortAsc, filter } = state.browser;


  let rows = filter

    ? entries.filter(e =>

        e.name.toLowerCase().includes(filter) ||

        e.topic.toLowerCase().includes(filter))

    : entries;


  rows = [...rows].sort((a, b) => {

    let av = a[sortCol], bv = b[sortCol];

    if (typeof av === 'string') av = av.toLowerCase();

    if (typeof bv === 'string') bv = bv.toLowerCase();

    if (av < bv) return sortAsc ? -1 : 1;

    if (av > bv) return sortAsc ?  1 : -1;

    return 0;

  });


  $browserStatus.textContent =

    `${rows.length} of ${entries.length} channel${entries.length !== 1 ? 's' : ''}`;


  // Update sort indicators

  document.querySelectorAll('#browser-table thead th[data-col]').forEach(th => {

    th.classList.toggle('sorted', th.dataset.col === sortCol);

    if (th.dataset.col === sortCol) {

      th.textContent = th.dataset.col.charAt(0).toUpperCase() + th.dataset.col.slice(1) +

        (sortAsc ? ' ▲' : ' ▼');

    } else {

      th.textContent = th.dataset.col.charAt(0).toUpperCase() + th.dataset.col.slice(1);

    }

  });


  $browserEmpty.style.display = rows.length === 0 ? 'block' : 'none';


  $browserTbody.innerHTML = rows.map(e => `

    <tr data-channel="${esc(e.name)}">

      <td class="td-name">${esc(e.name)}</td>

      <td class="td-users">${e.users}</td>

      <td class="td-topic">${esc(e.topic)}</td>

      <td class="td-join"><button class="join-btn" data-channel="${esc(e.name)}">Join</button></td>

    </tr>`).join('');

}
